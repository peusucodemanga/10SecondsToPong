2️⃣ Sistema de Legenda (Legenda.cs)

Atualizado para carregar e exibir mensagens de um arquivo .txt.

csharp
using UnityEngine;
using System.IO;
using System.Collections.Generic;

public class Legenda : MonoBehaviour
{
    public TextoLegenda TextoLegendaPrefab;
    public float duracao = 5.0f;
    private List<string> mensagens = new List<string>();

    float largura;
    float pixelPorSegundo;
    TextoLegenda textoAtual;

    void Start()
    {
        largura = GetComponent<RectTransform>().rect.width;
        pixelPorSegundo = largura / duracao;
    }

    public void CarregarMensagens(string nomeArquivo)
    {
        string caminho = Path.Combine(Application.streamingAssetsPath, nomeArquivo);

        if (File.Exists(caminho))
        {
            mensagens = new List<string>(File.ReadAllLines(caminho));
            Debug.Log($"Mensagens carregadas de: {nomeArquivo}");
        }
        else
        {
            Debug.LogError($"Arquivo não encontrado: {caminho}");
        }
    }

    void Update()
    {
        if (textoAtual == null || textoAtual.GetXPos <= -textoAtual.GetLargura)
        {
            if (mensagens.Count > 0)
            {
                string mensagemAleatoria = mensagens[Random.Range(0, mensagens.Count)];
                AdicionarTexto(mensagemAleatoria);
            }
        }
    }

    void AdicionarTexto(string texto)
    {
        textoAtual = Instantiate(TextoLegendaPrefab, transform);
        textoAtual.Inicializar(largura, pixelPorSegundo, texto);
    }
}

using UnityEngine;
using System.Collections.Generic;

public class Legenda : MonoBehaviour
{
    public TextoLegenda TextoLegendaPrefab;
    public float duracao = 5.0f; // Tempo que o texto leva para atravessar a tela

    private List<string> mensagens = new List<string>();
    private float largura;
    private float pixelPorSegundo;
    private TextoLegenda textoAtual;
    private Queue<string> filaMensagens = new Queue<string>(); // Fila de mensagens

    void Start()
    {
        largura = GetComponent<RectTransform>().rect.width;
        pixelPorSegundo = largura / duracao;
    }

    public void CarregarMensagens(string nomeArquivo)
    {
        TextAsset arquivo = Resources.Load<TextAsset>(nomeArquivo);

        if (arquivo != null)
        {
            mensagens = new List<string>(arquivo.text.Split('\n')); // Divide o arquivo em linhas
            Debug.Log($"Mensagens carregadas de: {nomeArquivo}");

            // Adiciona todas as mensagens na fila aleatoriamente
            EmbaralharMensagens();
            AdicionarTexto(filaMensagens.Dequeue()); // Exibe a primeira mensagem
        }
        else
        {
            Debug.LogError($"❌ Arquivo não encontrado: {nomeArquivo}");
        }
    }

    void Update()
    {
        // Quando o texto sai completamente da tela, já adicionamos outro
        if (textoAtual != null && textoAtual.GetXPos <= -textoAtual.GetLargura)
        {
            if (filaMensagens.Count == 0) // Se a fila esvaziar, embaralhamos novamente
                EmbaralharMensagens();

            string novaMensagem = filaMensagens.Dequeue();
            AdicionarTexto(" | " + novaMensagem); // Adiciona separador antes da nova mensagem
        }
    }

    void AdicionarTexto(string texto)
    {
        if (textoAtual == null)
        {
            textoAtual = Instantiate(TextoLegendaPrefab, transform);
        }
        textoAtual.Inicializar(largura, pixelPorSegundo, texto);
    }

    void EmbaralharMensagens()
    {
        List<string> tempMensagens = new List<string>(mensagens);
        while (tempMensagens.Count > 0)
        {
            int index = Random.Range(0, tempMensagens.Count);
            filaMensagens.Enqueue(tempMensagens[index]);
            tempMensagens.RemoveAt(index);
        }
    }
}

using UnityEngine;
using UnityEngine.UI;

public class EventoAleatorio : MonoBehaviour
{
    public float intervalo = 10f; 
    private Legenda legenda;
    private bool imagemTemaDesativada = false; // Controle para desativar a imagem apenas uma vez
    private Image imagemTema;

    void Start()
    {
        legenda = FindFirstObjectByType<Legenda>(); 
        legenda.CarregarMensagens("padrao");

        imagemTema = GameObject.Find("ImagemTema")?.GetComponent<Image>(); // Busca a imagem pelo nome

        InvokeRepeating("AtivarEvento", intervalo, intervalo);
    }

    void AtivarEvento()
    {
        if (!imagemTemaDesativada && imagemTema != null)
        {
            imagemTema.gameObject.SetActive(false); // Desativa a imagem
            imagemTemaDesativada = true; // Evita que ela seja desativada múltiplas vezes
        }

        int evento = Random.Range(0, 4);

        switch (evento)
        {
            case 0:
                Debug.Log("Evento: 1 bola extra");
                GameManager.instancia.CriarBolaGolf();
                break;
            case 1:
                Debug.Log("Evento: 2 bolas extras");
                GameManager.instancia.CriarBolaGolf();
                GameManager.instancia.Invoke("CriarBolaGolf", 1f);
                break;
            case 2:
                Debug.Log("Evento: Tela de TV");
                GameManager.instancia.TvBugada();
                break;
            case 3:
                Debug.Log("Evento: Pong Classico");
                SceneController.instancia.MudarParaScene(2, 9f);
                break;
        }
    }
}
